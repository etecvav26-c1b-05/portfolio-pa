#include <iostream>
using namespace std;

int main() {
    int numero;
    int potencia = 1;

    cout << "Digite um numero inteiro: ";
    cin >> numero;

    for (int i = 0; i <= numero; i++) {
        cout << "2^" << i << " = " << potencia << endl;

    
        potencia = potencia * 2;
    }

    return 0;
}

#include <iostream>
using namespace std;

int main() {
    int numero;
    int somaPares = 0;
    int somaImpares = 0;

    cout << "Digite um numero inteiro: ";
    cin >> numero;

    // Percorre de 0 até o número informado
    for (int i = 0; i <= numero; i++) {

        // Verifica se é par
        if (i % 2 == 0) {
            somaPares += i;
        } else {
            somaImpares += i;
        }
    }

    // Exibe os resultados
    cout << "Soma dos numeros pares: " << somaPares << endl;
    cout << "Soma dos numeros impares: " << somaImpares << endl;

    return 0;
}

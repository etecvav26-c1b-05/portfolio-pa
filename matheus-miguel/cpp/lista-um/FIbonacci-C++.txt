#include <iostream>
using namespace std;

int main() {
    int limite;

    cout << "Digite um numero entre 50 e 100: ";
    cin >> limite;

    // Validação do intervalo
    if (limite < 50 || limite > 100) {
        cout << "Numero invalido!" << endl;
        return 0;
    }

    int a = 0;
    int b = 1;
    int proximo;

    cout << "Sequencia de Fibonacci ate " << limite << ":" << endl;

    // Exibe os termos da sequência
    while (a <= limite) {
        cout << a << " ";

        proximo = a + b;
        a = b;
        b = proximo;
    }

    cout << endl;

    return 0;
}

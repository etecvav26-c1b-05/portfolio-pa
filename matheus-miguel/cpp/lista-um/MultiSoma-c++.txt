#include <iostream>
using namespace std;

int main() {
    int numero;
    int soma = 0;
    long long multiplicacao = 1;

    cout << "Digite os numeros:" << endl;

    // Continua até a soma ultrapassar 100
    while (soma <= 100) {
        cin >> numero;

        soma += numero;
        multiplicacao *= numero;
    }

    // Exibe o resultado final
    cout << "Multiplicacao dos numeros lidos: " << multiplicacao << endl;

    return 0;
}
